8   public interface SimpleStack {
9       /**
10       * Tests if this stack is empty
11       */
12      public boolean isEmpty();
13  
14      /**
15       * Returns the number of items in this stack.
16       */
17      public int getSize();
18  
19      /**
20       * Pushes an item onto the top of this stack.
21       * null item is allowed.
22       */
23      public void push(Item item);
24  
25      /**
26       * Looks at the object at the top of this stack without removing it from the stack.
27       */
28      public Item peek() throws EmptyStackException;
29  
30      /**
31       * Removes the object at the top of this stack and returns that object as the value of this function.
32       *
33       @throws
34       EmptyStackException if this stack is empty.
35       */
36      public Item pop() throws EmptyStackException;
37  }
